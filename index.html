int partition(int arr[],int l,int r) 
{ 
int p,i,j; 
p=arr[l]; 
i=l; 
j=r+1; 
while(i<j) 
{ 
// move i right 
do { 
i = i + 1; 
} while(i < r && arr[i] < p); 
// move j left 
do { 
j = j - 1; 
} while(j > l && arr[j] > p); 
if(i < j) 
swap(arr,i,j); 
} 
swap(arr,l,j); 
return j; 
}

{ 
int s;

{ 
s=partition(arr,l,r); 
quickSort(arr,l,s-1); 
quickSort(arr,s+1,r); 
} 
}

merge(a, b, low, mid, high);

{

b[i++]=a[j++];

for(k=j;k<=high;k++) 
b[i++]=a[k]; 
else 
for(k=h;k<=mid;k++) 
b[i++]=a[k]; 
for(k=low;k<=high;k++) 
a[k]=b[k]; 
}

#define MAX 100 
int parent[MAX]; 
int cost[MAX][MAX]; 
int tree[MAX][2]; 
/* Find function with path compression */ 
int find(int v) 
{ 
} 
if(parent[v] == -1) 
return v; 
return parent[v] = find(parent[v]); 
/* Union function */ 
void unionSets(int u, int v) 
{ 
parent[v] = u; 
} 
/* Kruskal's Algorithm */ 
void kruskal(int n) 
{ 
int i, j, k; 
int u, v; 
int min; 
int a = 0, b = 0; 
int totalCost = 0; 
for(k = 0; k < n - 1; k++) 
{ 
min = INF; 
/* Find minimum edge */ 
for(i = 0; i < n; i++) 
{ 
for(j = 0; j < n; j++) 
{ 
if(i != j && cost[i][j] < min) 
{ 
u = find(i); 
v = find(j); 
/* Check for cycle */ 
if(u != v) 
{ 
min = cost[i][j]; 
a = i; 
b = j; 
} 
} 
            } 
        } 
 
        /* Perform union */ 
        unionSets(find(a), find(b)); 
 
        tree[k][0] = a; 
        tree[k][1] = b; 
 
        totalCost += min; 
 
        /* Mark edge as visited */ 
        cost[a][b] = cost[b][a] = INF; 
    } 
 
    /* Display MST */ 
    printf("\nEdges in Minimum Spanning Tree:\n"); 
 
    for(i = 0; i < n - 1; i++) 
    { 
        printf("%d -> %d\n", 
               tree[i][0], 
               tree[i][1]); 
    } 
 
    printf("\nTotal Cost of MST = %d\n", 
           totalCost); 
}

Minimum Cost Spanning Tree (Prim's)
Design and implement Program to find the Minimum Cost Spanning Tree of a given connected undirected graph using Prim's algorithm.
{} C
#include <stdio.h> 
#define INF 999 
 
int prim(int c[10][10], int n, int s) 
{ 
    int visited[10], parent[10], dist[10]; 
    int i, j, u, min, cost = 0; 
 
    /* Initialization */ 
    for(i = 0; i < n; i++) 
    { 
        visited[i] = 0; 
        dist[i] = c[s][i]; 
        parent[i] = s; 
    } 
 
    visited[s] = 1; 
 
    /* Find n-1 edges */ 
    for(i = 0; i < n - 1; i++) 
    { 
        min = INF; 
 
        /* Find minimum distance vertex */ 
        for(j = 0; j < n; j++) 
        { 
            if(visited[j] == 0 && dist[j] < min) 
            { 
                min = dist[j]; 
                u = j; 
            } 
        } 
 
        visited[u] = 1; 
        cost = cost + dist[u]; 
 
        printf("Edge %d : %d -> %d  Cost = %d\n", 
               i + 1, parent[u], u, dist[u]); 
 
        /* Update distances */ 
        for(j = 0; j < n; j++) 
        { 
            if(visited[j] == 0 && c[u][j] < dist[j]) 
            { 
                dist[j] = c[u][j]; 
                parent[j] = u; 
            } 
        } 
    } 
 
    return cost; 
} 
 
int main() 
{ 
    int c[10][10]; 
    int n, s, i, j, totalCost; 
 
    printf("Enter number of vertices: "); 
    scanf("%d", &n); 
 
    printf("Enter adjacency matrix:\n"); 
 
    for(i = 0; i < n; i++) 
    { 
        for(j = 0; j < n; j++) 
        { 
            scanf("%d", &c[i][j]); 
        } 
    } 
 
    printf("Enter source vertex (0 to %d): ", n - 1); 
    scanf("%d", &s); 
 
    totalCost = prim(c, n, s); 
 
    printf("\nTotal cost of Minimum Spanning Tree = %d\n", 
           totalCost); 
 
    return 0; 
}

#include<stdio.h> 
 
#define INF 999 
 
void dijkstra(int c[10][10], int n, int s, int d[10]) 
{ 
    int v[10], min, u, i, j; 
 
    // Initialize distance and visited arrays 
    for(i = 1; i <= n; i++) 
    { 
        d[i] = c[s][i]; 
        v[i] = 0; 
    } 
 
    v[s] = 1; 
 
    // Find shortest path for all vertices 
    for(i = 1; i <= n; i++) 
    { 
        min = INF; 
 
        // Select the minimum distance vertex 
        for(j = 1; j <= n; j++) 
        { 
            if(v[j] == 0 && d[j] < min) 
            { 
                min = d[j]; 
                u = j; 
            } 
        } 
 
        v[u] = 1; 
 
        // Update distances 
        for(j = 1; j <= n; j++) 
        { 
            if(v[j] == 0 && (d[u] + c[u][j]) < d[j]) 
            { 
                d[j] = d[u] + c[u][j]; 
            } 
        } 
    } 
}

{

0/1 Knapsack Problem
Design and implement a Program to solve the 0/1 Knapsack problem using the Dynamic Programming method.
{} C
#include <stdio.h> 
 
int max(int a, int b) { 
    return (a > b) ? a : b; 
} 
 
int main() { 
    int n, capacity, i, j; 
    int w[100], p[100], F[100][100]; 
 
    printf("Enter the number of items: "); 
    scanf("%d", &n); 
 
    printf("Enter the capacity of knapsack: "); 
    scanf("%d", &capacity); 
 
    printf("Enter the profits (p) of the items:\n"); 
    for(i = 1; i <= n; i++) { 
        scanf("%d", &p[i]); 
    } 
 
    printf("Enter the weights (w) of the items:\n"); 
    for(i = 1; i <= n; i++) { 
        scanf("%d", &w[i]); 
    } 
 
    // Build the DP table F using bottom-up dynamic programming 
    for(i = 0; i <= n; i++) { 
        for(j = 0; j <= capacity; j++) { 
            if(i == 0 || j == 0) 
                F[i][j] = 0; 
            else if(w[i] <= j) 
                F[i][j] = max(F[i-1][j], p[i] + F[i-1][j - w[i]]); 
            else 
                F[i][j] = F[i-1][j]; 
        } 
    } 
 
    printf("Maximum profit = %d\n", F[n][capacity]); 
 
    // Traceback to find which items are included 
    printf("Items included (1-based index): "); 
    i = n; 
    j = capacity; 
    while(i > 0 && j > 0) { 
        if(F[i][j] != F[i-1][j]) { 
            printf("%d ", i);  // Item i is included 
            j -= w[i];         // Decrease remaining capacity 
        } 
        i--; // Move to previous item 
    } 
 
    printf("\n"); 
    return 0; 
}

#include <stdio.h> 
 
#define MAX 20 
 
int s[MAX], x[MAX], d, n; 
 
void sumofsub(int p, int k, int r) 
{ 
    int i; 
 
    x[k] = 1;  // Include current element 
 
    if (p + s[k] == d) 
    { 
        printf("Subset: "); 
        for (i = 1; i <= k; i++) 
        { 
            if (x[i] == 1) 
                printf("%d ", s[i]); 
        } 
        printf("\n"); 
    } 
    else if (k < n && (p + s[k] + s[k + 1] <= d)) 
    { 
        sumofsub(p + s[k], k + 1, r - s[k]); 
    } 
 
    // Exclude current element 
    if (k < n && (p + r - s[k] >= d) && (p + s[k + 1] <= d)) 
    { 
        x[k] = 0; 
        sumofsub(p, k + 1, r - s[k]); 
    } 
} 
 
int main() 
{ 
    int i, sum = 0; 
 
    printf("Enter the number of elements: "); 
    scanf("%d", &n); 
 
    printf("Enter the elements in increasing order:\n");

    { 
        scanf("%d", &s[i]); 
        sum += s[i]; 
    } 
 
    printf("Enter the target sum: "); 
    scanf("%d", &d); 
 
    if (sum < d || s[1] > d) 
    { 
        printf("No subset possible.\n"); 
    } 
    else 
    { 
        printf("Subsets whose sum is %d are:\n", d); 
        sumofsub(0, 1, sum); 
    }

}
Program 10
N Queen's Problem
Design and implement a Program for N Queens problem using Backtracking.
{} C
#include <stdio.h> 
#include <stdlib.h> 
 
int a[30], count = 0; 
 
/* Function to check whether queen can be placed */ 
int place(int pos) 
{ 
    int i; 
 
    for(i = 1; i < pos; i++) 
    { 
        if((a[i] == a[pos]) || 
           (abs(a[i] - a[pos]) == abs(i - pos))) 
            return 0; 
    } 
    return 1; 
} 
 
/* Function to print the solution */ 
void print_sol(int n) 
{ 
    int i, j; 
 
    count++; 
 
    printf("\nSolution #%d:\n", count); 
 
    for(i = 1; i <= n; i++) 
    { 
        for(j = 1; j <= n; j++) 
        { 
            if(a[i] == j) 
                printf("Q\t"); 
            else 
                printf("*\t"); 
        } 
        printf("\n"); 
    } 
} 
 
/* Backtracking function */ 
void queen(int n) 
{ 
    int k = 1; 
 
    a[k] = 0; 
 
    while(k != 0) 
    { 
        a[k]++; 
 
        while((a[k] <= n) && !place(k)) 
            a[k]++; 
 
        if(a[k] <= n) 
        { 
            if(k == n) 
                print_sol(n); 
            else 
            { 
                k++; 
                a[k] = 0; 
            } 
        } 
        else 
        { 
            k--; 
        } 
    } 
} 
int main() 
{ 
int n; 
printf("Enter the number of Queens: "); 
scanf("%d", &n); 
queen(n); 
printf("\nTotal number of solutions = %d\n", count); 
return 0; 
}
